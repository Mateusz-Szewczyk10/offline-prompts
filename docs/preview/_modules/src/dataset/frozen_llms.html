from .template.recipe import format_prompt

from ..types import Sentence

class LlamaFrozenLLMs(BaseFrozenLLMs):
    """Frozen LLMs for the recipe generation task.

    Note
    -------
    This model is available only on GPUs.

    Parameters
    -------
    model_id: str, default="meta-llama/Llama-2-7b-chat-hf"
        Model id of the Llama model.

    prompt_template: Callable, default=recipe.format_prompt
        Template function to format prompt.

    device: str, default="auto"
        Device.

    random_state: int, default=None
        Random state.

    """

    model_id: str = "meta-llama/Llama-2-7b-chat-hf"
    prompt_template: Callable = format_prompt
    device: str = "auto"
    random_state: Optional[int] = None

    def __post_init__(self):
        if self.random_state is not None:
            torch_seed(self.random_state, device=self.device)

        if self.device == "cpu":
            raise ValueError("device must not be cpu, but found False.")

        self.tokenizer = LlamaTokenizer.from_pretrained(self.model_id)
        self.model = LlamaForCausalLM.from_pretrained(
            self.model_id,
            load_in_8bit=True,
            device_map=self.device,
            torch_dtype=torch.float16,
        )

    def _generate_output_sentence(
        self,
        query: Sentence,
        prompt: Sentence,
    ):
        """Generate output sentence given query and prompt.

        Parameters
        -------
        query: Sentence, shape (n_samples, )
            Qurey specified by users.

        prompt: Sentence, shape (n_samples, ) or (n_samples, n_actions, ) or (n_actions, )
            Discrete prompts for each given user context and query.

        """
        prompt = self.prompt_template(query=query, prompt=prompt)
        input = self.tokenizer(prompt, return_tensors="pt")

        with torch.no_grad():
            output_sentence = self.tokenizer.decode(
                self.model.generate(**input, max_new_tokens=100)[0],
                skip_special_tokens=True,
            )
        return output_sentence
